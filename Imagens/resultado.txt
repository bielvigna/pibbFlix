import java.util.*;

public class Main {

    public static void main(String[] args) {
        // Definir produtos
        ProdutoEstoque prod1 = new ProdutoEstoque("prod-001", 5);
        ProdutoEstoque prod2 = new ProdutoEstoque("prod-002", 15);
        ProdutoEstoque prod3 = new ProdutoEstoque("prod-003", 20);

        // Criar mapa de produtos (estoque)
        Map<String, Integer> produtos = new HashMap<>();
        produtos.put(prod1.getCodigoProduto(), prod1.getQuantidade());
        produtos.put(prod2.getCodigoProduto(), prod2.getQuantidade());
        produtos.put(prod3.getCodigoProduto(), prod3.getQuantidade());

        // Criar fila de pedidos (como texto)
        Queue<String> fila = new LinkedList<>();

        fila.offer("Pedido 101: (item: " + prod1.getCodigoProduto() + ", Qtd: 5), (item: " + prod2.getCodigoProduto() + ", Qtd: 2)");
        fila.offer("Pedido 102: (item: " + prod2.getCodigoProduto() + ", Qtd: 10), (item: " + prod3.getCodigoProduto() + ", Qtd: 8)");
        fila.offer("Pedido 103: (item: " + prod1.getCodigoProduto() + ", Qtd: 2), (item: " + prod3.getCodigoProduto() + ", Qtd: 10)");

        // Exibir fila
        System.out.println("Fila de pedidos:");
        for (String pedido : fila) {
            System.out.println(pedido);
        }

        // Exibir estoque inicial
        System.out.println("\nEstoque inicial:");
        for (Map.Entry<String, Integer> entry : produtos.entrySet()) {
            System.out.println(entry.getKey() + " -> " + entry.getValue());
        }
    }
}

public class ProdutoEstoque {
   
    private String codigoProduto;
 
    private int quantidade;
 
    public ProdutoEstoque(String codigoProduto, int quantidade) {
        this.codigoProduto = codigoProduto;
        this.quantidade = quantidade;
    }
 
    public String getCodigoProduto() {
        return codigoProduto;
    }
 
    public int getQuantidade() {
        return quantidade;
    }
 
    public void setQuantidade(int quantidade) {
        this.quantidade = quantidade;
    }

        Map<Integer, String> resultado = new HashMap<>();

           while (!filaPedidos.isEmpty()) {

            boolean atendimento = true;

            for (ItemPedido item : pedido.getListaDeItens()) {

public class ItemPedido {
   
    private String codigoProduto;
   
    private int quantidadePedida;
 
    public ItemPedido(String codigoProduto, int quantidadePedida) {
        this.codigoProduto = codigoProduto;
        this.quantidadePedida = quantidadePedida;
    }
 
    public String getCodigoProduto() {
        return codigoProduto;
    }
   
    public int getQuantidadePedida() {
        return quantidadePedida;
    }
 
    public void setQuantidadePedida(int quantidadePedida) {
        this.quantidadePedida = quantidadePedida;
    }

    
    public String toString() {
        return " ////////// ItemPedido ////////// " + "\nCodigo produto = " + codigoProduto + "\nQuantidade pedida = " + quantidadePedida + "\n /////////////////////////////////// ";        }
    
}
